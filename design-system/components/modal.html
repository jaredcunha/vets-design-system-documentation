Modal

AdditionalInfo components are interactive components that reveal and hide content from the page.

import React from 'react';
import Modal from './Modal';

export default function ModalExample(props) {
  return (
    <Modal
      title={props.title}
      id={props.id}
      status={props.status}
      visible
      onClose={() => {}}>
      {props.content}
    </Modal>
  );
}

<script type="text/javascript">
  function showModal(elem) {
    var parentClasses = document.getElementById("modal-crisisline").classList;
    if(elem.classList.contains('va-overlay-close')) {
      parentClasses.add('va-overlay');
    } else if(parentClasses.contains('va-overlay')) {
      parentClasses.remove('va-overlay');
    }
  }
</script>
